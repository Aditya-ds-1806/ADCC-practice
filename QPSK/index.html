Input: NO_OF_BITS, EB_N0_DB, Bk
Output: BER

map([bit1, bit2]) = {
    00: [1/sqrt(2), 1/sqrt(2)],
    01: [1/sqrt(2), -1/sqrt(2)],
    10: [-1/sqrt(2), 1/sqrt(2)],
    11: [-1/sqrt(2), -1/sqrt(2)]
}

ML([bit1, bit2]) = {
    00: bit1 > 0 and bit2 > 0
    01: bit1 > 0 and bit2 < 0
    10: bit1 < 0 abd bit2 > 0
    11: bit1 < 0 and bit2 < 0
}

Sk = map(Bk.group(2))

for EB_N0 in EB_N0_DB
    Nk = AWGN(EB_N0, 2D)
    Yk = Sk + Nk
    sHat = ML(Yk)
    unchangedSymbols = count(sHat === Sk)
    BER = 1 - (unchangedSymbols/NO_OF_BITS)

plot(BER, BER_THEORETICAL)
